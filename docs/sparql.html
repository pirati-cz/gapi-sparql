        name: 'gapi',
        overwrite: true,
        mongoDomain: 'localhost',
        mongoPort: 27017
      }, function(store) {});
    }

    Sparql.prototype.query = function(query, callback) {
      if (callback == null) {
        callback = null;
      }
      return this.store.execute(query, function(success, results) {
        if (callback === null) {
          callback();
        }
        return results;
      });
    };

    return Sparql;

  })();
